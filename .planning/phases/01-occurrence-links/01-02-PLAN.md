```typescript
private async hydrateFromOccurrenceId(id: string): Promise<void> {
  const {data: occurrence} = await supabase()
    .from('occurrences')
    .select()
    .eq('id', id)
    .maybeSingle<Occurrence>();
  if (!occurrence) return; // not found — silent fallback per decisions

  const date = dateFromObservedAt(occurrence.observed_at);
  // Bypass the date setter to avoid writing ?d= to history
  this.#date = date;
  await this.fetchOccurrences(date);

  // Center map on occurrence location
  const {lon, lat} = occurrence.location as {lon: number; lat: number};
  const [x, y] = fromLonLat([lon, lat]);
  this.mapRef.value!.setView(x, y, 12, {skipEvent: true});
  this.focusedOccurrenceId = id;
}
```

CRITICAL ANTI-PATTERNS:
- Do NOT use `this.date = date` — the setter calls `setQueryParams({d})` and pollutes browser history
- Do NOT call `setView` in `connectedCallback` — `mapRef.value` is undefined there; only valid in/after `firstUpdated`
- Do NOT use `skipEvent: false` for `setView` — that fires 'map-move' which writes `?x=/?y=/?z=` to history
- Coordinate order: `fromLonLat([lon, lat])` not `fromLonLat([lat, lon])`
